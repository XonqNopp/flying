\begin{CheckListRule}{Before starting engine}{}%% {{{
	\PreflightInspectionCOMPLETED
	\WheelChocksAndPitotCoverOnBoard
	\ParkingBrakeTESTED
	\PassengerBriefingCOMPLETED
	\SeatsADJUSTED
	\CowlFlaps{OPEN}
	\AvionicsMasterSwitch{OFF}
	\ElectricalOFF
	\CircuitBreakers{2 rows IN}
	\MasterSwitch{ON}
	\HourMeterStandWRITTEN
	\FuelQuantityLR
	\FuelSelectorValveBOTH
\end{CheckListRule}%% }}}
\begin{CheckList}{Starting engine}{}%% {{{
	\Check{BCN + NAV lights, FLARM, Aspen PFD}{ALL ON}
	\PropellerMAX
	\ThrottleFrictionLockADJUSTED
	\Check{Throttle}{5mm}
	\Mixture{CUT OFF}
	\begin{ConditionalChecks}{If engine cold:}
		\Check{Auxiliary fuel pump}{ON}
		\Mixture{RICH UP TO 3--5 GPH then CUT OFF}
		\Check{Auxiliary fuel pump}{OFF}
	\end{ConditionalChecks}
	\PropellerAreaCLEAR
	\IgnitionKeySTART
	\Check[hh]{Mixture}{SMOOTHLY TO RICH when engine fires}
	\Check[hh]{Engine warm up}{\RPM{1000--1200}}
	\Check[hh]{Oil pressure}{}
	\Check[hh]{Starter}{CHECK DISCONNECTED (light + ammeter)}
	\AvionicsMasterSwitch{ON}
	\Check{Radios}{ON, LIGHTS TESTED, SET}
	\WingFlapsCHECKED
\end{CheckList}%% }}}
\begin{CheckListRule}{Taxi}{}%% {{{
	\BlockTime{off}
	\TaxiLandingLights{ON}
	\ParkingBrake{RELEASED}
	\BrakesCHECK
	\NoseWheelSteeringCHECK
	\FlightInstrumentsCHECK
	\Check[hh]{Transponder}{GND}
\end{CheckListRule}%% }}}

\begin{CheckListRule}{Engine check}{}%% {{{
	\Check{PAX checklist}{GIVEN}
	\ParkingBrake{SET (pedals depressed)}
	\TaxiLandingLights{OFF}
	\ThrottleRpm{1800}
	\Check{Magnetos}{CHECKED (max drop \RPM{150})}
	\Check{Propeller}{CYCLE, then MAX RPM}
	\Check{Mixture, EGT and Fuel Flow}{}
	\Check{Ammeter}{}
	\OilPressureTemperature
	\Check{Annunciator panel}{TESTED then NO LIGHTS}
	\ThrottleRpm[idle]{500--700}
	\EngineWarmUp{1000}
\end{CheckListRule}%% }}}
