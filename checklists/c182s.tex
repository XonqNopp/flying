\begin{CheckListRule}{Before starting engine}{}% {{{
	% PERSONAL: first check the version of the official on-board checklist
	\CheckChecklist

	\PreflightInspectionCOMPLETED
	\WheelChocksAndPitotCoverREMOVED
	\Check{Seats, seat belts}{ADJUSTED and LOCKED}
	\ParkingBrakeTESTED
	\PassengerBriefingCOMPLETED
	\Avionics{OFF}
	\Check{Electrical equipment}{OFF}
	\CowlFlaps{OPEN}
	\CircuitBreakers{2 rows IN}
	\MasterSwitch{ON}
	\HourMeterWRITTEN
	\FuelQuantityLR
	\FuelSelectorBOTH
\end{CheckListRule}% }}}
\begin{CheckList}{Starting engine}{}% {{{
	\Check{BCN + NAV lights, FLARM, Aspen PFD}{ALL ON}
	\PropellerHIGH

	% PERSONAL: better to check throttle friction lock before taxi
	\PERSONAL{\ThrottleFrictionLockADJUSTED}

	% PERSONAL: use SI unit
	\Check{Throttle}{\PERSONAL{5mm}}

	\MixtureIDLE

	\begin{ConditionalChecks}{If engine cold:}% {{{
		\AuxFuelPump{ON}

		% PERSONAL: ensure we go full rich and not smoothly
		\Mixture{ADVANCE \PERSONAL{RICH} up to 3--5 GPH then IDLE CUT OFF}

		\AuxFuelPump{OFF}
	\end{ConditionalChecks}% }}}

	% No-read section {{{
	\PropellerAreaCLEAR

	% PERSONAL: there is an indicator light to show the starter is running.
	% We can check it afterwards to see the starter really disconnected.
	% But to trust it, we have to check right when we make contact that it is indeed working properly.
	\Check[hh]{Magnetos}{START \PERSONAL{and check light}\ (release when engine starts)}

	\Check[hh]{Mixture}{ADVANCE smoothly to RICH when engine fires}
	\Check[hh]{Engine warm up}{\RPM{1000--1200}}
	\Check[hh]{Oil pressure}{}

	% PERSONAL: check indicator light too
	\Check[hh]{Starter}{CHECK DISCONNECTED (\PERSONAL{light +} ammeter)}
	% }}}

	% PERSONAL: we turn on avionics, the master switch we only toggle
	\Check{\PERSONAL{Avionics}}{ON}

	% PERSONAL: test radio lights before going further
	\Check{Radios}{ON, \PERSONAL{LIGHTS TESTED}, SET}

	\WingFlapsCHECK
\end{CheckList}% }}}
\vfill
\begin{CheckList}{Taxi}{}% {{{
	% PERSONAL: write down block-off time
	\ItemZero\PERSONAL{\BlockTime{off}}

	\TaxiLandingLights{ON}
	\ParkingBrake{RELEASED}

	% No-read section {{{
	% PERSONAL: we not only check the brake pedals, but the brakes themselves
	\Check[h]{\PERSONAL{Brakes}}{}

	\Check[hh]{Nose wheel steering}{}

	% PERSONAL: we do not SET anything right now, we only check
	\Check[hh]{Flight instruments}{\PERSONAL{CHECK}}

	\Check[hh]{Transponder}{GND (Mode S)}
	% }}}
\end{CheckList}% }}}

\begin{CheckListRule}{Engine check}{}% {{{
	% PERSONAL: provide PAX checklist so they have something to do
	\ItemZero\PERSONAL{\Check{PAX checklist}{GIVEN}}

	\ParkingBrake{SET, brake pedals depressed}
	\TaxiLandingLights{OFF}
	\ThrottleRpm{1800}
	\Magnetos{CHECKED (max drop \RPM{150})}
	\Check{Propeller}{CYCLE then HIGH RPM}
	\Check{Mixture, EGT and Fuel Flow}{}
	\Check{Ammeter}{}
	\OilPressureTemperature

	% PERSONAL: we do not check the lights, we test them
	\Check{Annunciator panel}{\PERSONAL{TESTED} then ensure no annunciator illuminated}

	\ThrottleRpm[idle]{500--700}
	\Throttle{1000}

	% SKIP: done in BeforeStartingEngine
	\SKIP{\ThrottleFrictionLockADJUSTED}
\end{CheckListRule}% }}}
