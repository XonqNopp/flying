\begin{CheckListRule}{Before starting engine}{}%% {{{
	\Check{Preflight inspection        }{COMPLETED}
	\Check{Wheel chocks and pitot cover}{REMOVED, ON BOARD}
	\Check{Passenger briefing          }{COMPLETED}
	\Check{Seats, seat belts           }{ADJUSTED and LOCKED}
	\Check{Parking brake               }{TESTED and SET}
	\Check{Avionics master switch      }{OFF}
	\Check{Electrical equipment        }{OFF}
	\Check{Circuit breakers            }{2 rows IN}
	\Check{Master switch               }{ON}
	\Check{Hour meter stand            }{WRITTEN}
	\Check{Fuel quantity L and R       }{}
	\Check{Fuel selector valve         }{BOTH}
	\Check{Fuel shutoff valve          }{ON (full IN)}
\end{CheckListRule}%% }}}
\begin{CheckList}{Starting engine}{}%% {{{
	\Check{Beacon light, FLARM, ASPEN}{ALL ON}
	\Check{Throttle friction lock    }{ADJUSTED}
	\Check{Throttle                  }{\nicefrac{1}{4}''}
	\Check{Mixture                   }{CUT OFF}
	\begin{ConditionalChecks}{If engine cold:}
		\Check{Auxiliary fuel pump}{ON}
		\Check{Mixture            }{UP TO 3--5GPH then CUT OFF}
		\Check{Auxiliary fuel pump}{OFF}
	\end{ConditionalChecks}
	\Check[h]{Propeller area         }{CLEAR}
	\Check[hh]{Ignition key          }{START}
	\Check[hh]{Mixture               }{SMOOTHLY TO RICH}
	\Check[hh]{Engine warm up        }{\RPM{1000--1200}}
	\Check[hh]{Oil pressure          }{}
	\Check[hh]{Starter               }{CHECK DISCONNECTED (ammeter)}
	\Check{Avionics master switch    }{ON}
	\Check{FLARM lights              }{}
	\Check{Radios                    }{ON, SET}
	\Check{Position light            }{AS REQUIRED}
	\Check{Wing flaps                }{CHECKED then RETRACTED}
\end{CheckList}%% }}}
\SkipPage

\end{CheckListRule}%% }}}
