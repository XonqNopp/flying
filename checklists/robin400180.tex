\begin{CheckListRule}{Engine start}{}% {{{
	\Lights{NAV}{ON}
	\Check[h]{Fuel pump}{ON}
	\Check[hh]{Fuel pressure}{CHECKED MOVING}
	\Magnetos{LEFT}
	\begin{ConditionalChecks}{If WEATHER cold:}
		\Throttle{10x open-close then $5mm$}
		\Magnetos{BOTH}
		\StarterENGAGE
		\Throttle{5-6x open-close then idle}
		\Magnetos{LEFT}
		\StarterENGAGE
		\Throttle{Increase progressively}
	\end{ConditionalChecks}
	\Throttle{3x open-close then $5mm$}
	\PropellerAreaCLEAR
	\Check[hh]{Starter}{ENGAGE, TIME CHECK}
	\Check[hh]{Magnetos}{BOTH}
	\Check[hh]{Throttle}{\RPM{1200}}
	\Check[hh]{Oil pressure}{MIN YELLOW ARC (max 30s)}
\end{CheckListRule}% }}}
