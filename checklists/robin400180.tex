\begin{CheckListRule}{Engine start}{}% {{{
	\Lights{NAV}{ON}
	\Check[h]{Fuel pump}{ON}

	% PERSONAL: check the fuel pressure is moving when we turn pump ON
	% Rationale: it is the only time we can check the pressure indicator is working properly
	\PERSONAL{\Check[hh]{Fuel pressure}{CHECKED MOVING}}

	\Magnetos{LEFT}

	% Cold weather priming according to Daniel Berchtold {{{
	\begin{ConditionalChecks}{If WEATHER cold:}
		\PERSONAL{\Throttle{10-12x open-close then $5mm$}}
		\PERSONAL{\Magnetos{BOTH}}
		\PERSONAL{\StarterENGAGE}
		\PERSONAL{\Throttle{5-6x open-close then idle (no flame return)}}
		\PERSONAL{\Magnetos{LEFT}}
		\PERSONAL{\StarterENGAGE}
		\PERSONAL{\Throttle{Increase progressively}}
	\end{ConditionalChecks}% }}}

	% Regular priming according to Daniel Berchtold
	\PERSONAL{\Throttle{3x open-close then $5mm$}}

	% No-read section {{{
	\PropellerAreaCLEAR
	\Check[hh]{Starter}{ENGAGE, TIME CHECK}
	\Check[hh]{Magnetos}{BOTH}
	\Check[hh]{Throttle}{\RPM{1200}}
	\Check[hh]{Oil pressure}{MIN YELLOW ARC (after max 30s)}
	% }}}
\end{CheckListRule}% }}}
