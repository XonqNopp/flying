\begin{CheckListRule}{Before starting engine}{}%% {{{
	\CheckChecklist{\SrcName\ \SrcVersion}

	\Check{Pedals}{ADJUSTED and LOCKED}
	\Check{Seat belts}{FASTENED}

	\Check{Canopy}{CLOSED and LOCKED}
	\Check{Passenger briefing}{COMPLETED}
	\Check{Friction lock}{ADJUSTED}
	\Check{Avionics master}{OFF}

\begin{CheckList}{Starting engine}{}%% {{{
	\Check{Position lights}{ON}

	\FuelPump{ON (ticking)}

	\Check{Propeller}{FULL RPM}
	\Check{Throttle}{IDLE        \textbf{if engine cold --- else} 5mm\hspace{18mm}.}
	\Check{Choke}{   ON (pulled) \textbf{if engine cold --- else} OFF (pushed)}
	\Check{Propeller area}{FREE}
	\Check[h]{Ignition key}{START}
	\Check[hh]{Engine warm up}{\RPM{1000--1200}}
	\Check[hh]{Oil pressure}{SECTOR GREEN (max 10s)}
	\Check{Choke}{PUSHED}
	\FuelPump{OFF}
	\Check{Warning lights \emph{Generator, Fuel pressure, Low voltage}}{OFF}
	\Check{Avionics}{ON}
	\Check{Radios, GPS}{ON, SET}
	\Check{Flaps}{Checked then retracted}
	\Check{Position lights}{AS REQUIRED (max 50\% OBT)}
\end{CheckList}%% }}}
